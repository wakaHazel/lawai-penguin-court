# 企鹅法庭文游核心工作流实施计划

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** 在现有案件录入、模拟、分析底座上，落地第一版“14 节点民事庭审文游引擎”，让系统可以按法定阶段推进、记录关键分叉、生成结构化复盘，并驱动前端沉浸式庭审舞台页。

**Architecture:** 保留现有 `FastAPI + SQLite + React` 主体结构，不推翻现有 `/simulate`、`/analysis` 路由，而是在后端新增“工作流目录 + 状态规则 + 事件池 + TrialRun 持久化”四层。前端继续沿用当前工作台与庭审页，但把单段场景文本升级为“CG 文本主舞台 + 右侧态势栏 + 底部策略卡”的固定舞台布局，元器仅消费结构化结果，不承担主链判定。

**Tech Stack:** Python 3.11, FastAPI, Pydantic, SQLite, React 18, TypeScript, Vite。

**Repo Note:** 当前 `E:\lawai` 不是 git 仓库。执行本计划时，原本的 “Commit” 步骤统一替换为“本地检查点记录”；如果后续初始化 git，再按本计划给出的提交信息执行。

## 范围锁定

本计划只覆盖已批准规格中的第一优先级内容：

- `N01 -> N14` 主干节点跑通
- 六大法定阶段顺序固定
- `N07`、`N08`、`N10`、`N11` 四个关键分叉点可感知
- 7 个隐藏状态生效
- TrialRun / RunCheckpoint / 历史回放结构落地
- 复盘报告结构化输出
- Web 端庭审舞台页完成核心渲染

本计划明确不做：

- 开放式自由聊天庭审
- 第二批案件类型扩展
- 元器主链编排
- 立法沙盘模块扩容
- 复杂动画、音视频和公网部署

### 职责划分

- `trial_workflow.py`: 定义节点、选项、隐藏状态、TrialRun、Checkpoint 的统一结构。
- `workflow_catalog.py`: 固定 14 节点目录、四类事件池、四个关键回放点。
- `workflow_rules.py`: 负责初始值、案件类型修正、选项效果模板、状态裁剪。
- `workflow_renderer.py`: 把结构化节点渲染成 `cg_caption / court_progress / pressure_shift / choice_prompt`。
- `trial_workflow_engine.py`: 负责 start / advance / resume 三类主流程。
- `trial_run_repository.py` 与 `checkpoint_repository.py`: 持久化 TrialRun、回放点、历史推进轨迹。
- `simulation.py`: 保持路由入口不变，但改为调用新引擎。
- `analysis.py`: 从 TrialRun 历史和隐藏状态生成更像“文游结算”的分析结果。
- Web 侧 `types + services + controller + components`: 把旧的单段文字模拟页升级成结构化舞台页。

```python
def save_trial_run(run: TrialRunSnapshot) -> TrialRunSnapshot:
    with get_connection() as connection:
        connection.execute(
            """
            INSERT INTO trial_runs (
                trial_run_id, case_id, current_node_id, current_stage, turn_index, payload_json
            )
            VALUES (?, ?, ?, ?, ?, ?)
            ON CONFLICT(trial_run_id) DO UPDATE SET
                current_node_id = excluded.current_node_id,
                current_stage = excluded.current_stage,
                turn_index = excluded.turn_index,
                payload_json = excluded.payload_json,
                updated_at = CURRENT_TIMESTAMP
            """,
            (
                run.trial_run_id,
                run.case_id,
                run.current_node_id,
                run.current_stage.value,
                run.turn_index,
                run.model_dump_json(),
            ),
        )
    return run.model_copy(deep=True)


def get_trial_run(trial_run_id: str) -> TrialRunSnapshot | None:
    with get_connection() as connection:
        row = connection.execute(
            "SELECT payload_json FROM trial_runs WHERE trial_run_id = ?",
            (trial_run_id,),
        ).fetchone()
    return None if row is None else TrialRunSnapshot.model_validate_json(row["payload_json"])


def get_latest_trial_run(case_id: str) -> TrialRunSnapshot | None:
    with get_connection() as connection:
        row = connection.execute(
            "SELECT payload_json FROM trial_runs WHERE case_id = ? ORDER BY updated_at DESC LIMIT 1",
            (case_id,),
        ).fetchone()
    return None if row is None else TrialRunSnapshot.model_validate_json(row["payload_json"])

        row = connection.execute(

            SELECT payload_json
            FROM simulation_turns
            WHERE trial_run_id = ?
            ORDER BY turn_index DESC

    return [SimulationSnapshot.model_validate_json(row["payload_json"]) for row in rows]
```

```python
BASELINE_STATE = HiddenStateSnapshot()

CASE_TYPE_MODIFIERS = {
    CaseType.LABOR_DISPUTE: {"evidence_strength": -8, "opponent_pressure": 4, "contradiction_risk": 8, "surprise_exposure": 8, "settlement_tendency": 4},
    CaseType.PRIVATE_LENDING: {"evidence_strength": 6, "contradiction_risk": -4, "surprise_exposure": 4, "settlement_tendency": -6},
}

EFFECT_TEMPLATES = {
    "procedure_objection": {"procedure_control": 8, "judge_trust": 4, "surprise_exposure": -8},
    "direct_explanation": {"judge_trust": 8, "contradiction_risk": -8, "surprise_exposure": -4},
    "evidence_reinforcement": {"evidence_strength": 12, "procedure_control": 4, "surprise_exposure": -8},
}


def build_initial_state(case_type: CaseType) -> HiddenStateSnapshot:
    state = BASELINE_STATE.model_copy(deep=True)
    for key, delta in CASE_TYPE_MODIFIERS.get(case_type, {}).items():
        setattr(state, key, max(0, min(100, getattr(state, key) + delta)))
    return state


def apply_effect_template(state: HiddenStateSnapshot, effect_key: str) -> HiddenStateSnapshot:
    next_state = state.model_copy(deep=True)
    for key, delta in EFFECT_TEMPLATES[effect_key].items():
        setattr(next_state, key, max(0, min(100, getattr(next_state, key) + delta)))
    return next_state
```

```python
    return next(node for node in get_civil_trial_workflow().nodes if node.node_id == node_id)


def resolve_choice(node_id: str, choice_id: str) -> WorkflowChoice:
    node = get_node(node_id)
    return next(choice for choice in node.choices if choice.choice_id == choice_id)


def is_checkpoint_node(node_id: str) -> bool:
    return node_id in get_civil_trial_workflow().checkpoint_node_ids


def build_checkpoint(run: TrialRunSnapshot) -> RunCheckpointSnapshot:
    return RunCheckpointSnapshot(
        checkpoint_id=f"CP-{run.current_node_id}",
        trial_run_id=run.trial_run_id,
        source_node_id=run.current_node_id,
        turn_index=run.turn_index,
        payload_json=run.model_dump_json(),
    )


def build_snapshot(case_profile: CaseProfile, node: WorkflowNodeDefinition, state: HiddenStateSnapshot, turn_index: int) -> SimulationSnapshot:
    rendered = render_scene(node=node, event=None, state=state)
    return SimulationSnapshot(
        simulation_id=f"sim_{case_profile.case_id.lower()}",
        case_id=case_profile.case_id or "",
        current_stage=node.stage,
        turn_index=turn_index,
        node_id=node.node_id,
        scene_title=node.title,
        scene_text=rendered["court_progress"],
        cg_caption=rendered["cg_caption"],
        court_progress=rendered["court_progress"],
        pressure_shift=rendered["pressure_shift"],
        choice_prompt=rendered["choice_prompt"],
        hidden_state_summary={"procedure_control": "占优" if state.procedure_control >= 60 else "中性"},
        available_actions=[choice.label for choice in node.choices],
    )


def start_trial_run(case_profile: CaseProfile) -> tuple[TrialRunSnapshot, SimulationSnapshot]:
    state = build_initial_state(case_profile.case_type)
    node = get_civil_trial_workflow().nodes[0]
    snapshot = build_snapshot(case_profile=case_profile, node=node, state=state, turn_index=1)
    run = TrialRunSnapshot(trial_run_id=snapshot.simulation_id, case_id=snapshot.case_id, current_node_id=node.node_id, current_stage=node.stage, turn_index=1, state=state)
    return run, snapshot


def advance_trial_run(case_profile: CaseProfile, current_run: TrialRunSnapshot, selected_choice_id: str) -> tuple[TrialRunSnapshot, SimulationSnapshot, RunCheckpointSnapshot | None]:
    choice = resolve_choice(current_run.current_node_id, selected_choice_id)
    next_state = apply_effect_template(current_run.state, choice.effect_key)
    checkpoint = build_checkpoint(current_run) if is_checkpoint_node(current_run.current_node_id) else None
    next_node = get_node(choice.next_node_id)
    next_snapshot = build_snapshot(case_profile=case_profile, node=next_node, state=next_state, turn_index=current_run.turn_index + 1)
    return current_run.model_copy(update={"current_node_id": next_node.node_id, "current_stage": next_node.stage, "turn_index": current_run.turn_index + 1, "state": next_state}), next_snapshot, checkpoint


def resume_trial_run(case_profile: CaseProfile, checkpoint: RunCheckpointSnapshot) -> tuple[TrialRunSnapshot, SimulationSnapshot]:
    run = TrialRunSnapshot.model_validate_json(checkpoint.payload_json)
    node = get_node(run.current_node_id)
    snapshot = build_snapshot(case_profile=case_profile, node=node, state=run.state, turn_index=run.turn_index)
    return run, snapshot
```
