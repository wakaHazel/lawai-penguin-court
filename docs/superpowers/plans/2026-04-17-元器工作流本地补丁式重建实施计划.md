    patched = clone(workflow)

        "    def parse_list(text, default_value):\n"
        "        raw = str(text or '').strip()\n"
        "        if not raw:\n"
        "            return [default_value]\n"
        "        try:\n"
        "            parsed = json.loads(raw)\n"
        "            if isinstance(parsed, list):\n"
        "                cleaned = [str(x).strip() for x in parsed if str(x).strip()]\n"
        "                if cleaned:\n"
        "                    return cleaned\n"
        "        except Exception:\n"
        "            pass\n"
        "        cleaned = [item.strip() for item in raw.split(',') if item.strip()]\n"
        "        return cleaned or [default_value]\n"
        "    return {\n"
        "        'issues_out': parse_list(params.get('issues_in'), '争议焦点待补充'),\n"
        "        'keywords_out': parse_list(params.get('keywords_in'), '案件关键词待补充'),\n"

```python
def copy_source_package() -> None:
    if OUTPUT_DIR.exists():
        shutil.rmtree(OUTPUT_DIR)
    shutil.copytree(SOURCE_DIR, OUTPUT_DIR)


def write_json(path: Path, payload: dict) -> None:
    path.write_text(json.dumps(payload, ensure_ascii=False, separators=(",", ":")), encoding="utf-8")


def zip_output_dir() -> None:
    if OUTPUT_ZIP.exists():
        OUTPUT_ZIP.unlink()
    with zipfile.ZipFile(OUTPUT_ZIP, "w", zipfile.ZIP_DEFLATED) as zf:
        for file_path in sorted(OUTPUT_DIR.rglob("*")):
            if file_path.is_file():
                zf.write(file_path, file_path.relative_to(OUTPUT_DIR))


def main() -> None:
    copy_source_package()
    w00_path = OUTPUT_DIR / "0948102d-85b6-4ab5-a29f-a25be03ac178_workflow.json"
    w02_path = OUTPUT_DIR / "9e474aaa-7d9a-4f60-8dd4-b51abf4a95a4_workflow.json"
    w04_path = OUTPUT_DIR / "aaaab6c3-f250-4c73-ab68-9da834ea737d_workflow.json"

    write_json(w00_path, patch_w00_workflow(json.loads(w00_path.read_text(encoding='utf-8'))))
    write_json(w02_path, patch_w02_workflow(json.loads(w02_path.read_text(encoding='utf-8'))))
    write_json(w04_path, patch_w04_workflow(json.loads(w04_path.read_text(encoding='utf-8'))))
    zip_output_dir()
```
